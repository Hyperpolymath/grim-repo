# ai.txt - AI Training and Usage Policies
# Specification: https://site.spawning.ai/ai-txt

# This file specifies how AI systems may interact with this project's content.

# === TRAINING DATA USAGE ===

# Allow AI training on public documentation
ai-training: documentation=allow

# Allow AI training on code (respects dual MIT + Palimpsest v0.8 license)
ai-training: code=allow

# Require attribution in training datasets
ai-training: attribution=required

# Allow AI training on test data
ai-training: tests=allow

# === RESTRICTIONS ===

# Prohibit training on private/internal discussions (if any)
ai-training: private-discussions=disallow

# Prohibit training on personally identifiable information
ai-training: pii=disallow

# Prohibit training on security vulnerabilities (before public disclosure)
ai-training: embargoed-security=disallow

# === ATTRIBUTION REQUIREMENTS ===

# When using this project's content in AI training, please:
# 1. Attribute to "GrimRepo Contributors"
# 2. Link to: https://gitlab.com/extensions-library/monkey-scripts/grimrepo-scripts
# 3. Note dual license: MIT + Palimpsest v0.8
# 4. Respect emotional safety provisions of Palimpsest License

# === AI-GENERATED CONTRIBUTIONS ===

# AI-assisted contributions are WELCOME under these conditions:
# 1. Humans review and take responsibility for AI-generated code
# 2. Disclose AI usage in merge request descriptions (optional but appreciated)
# 3. Ensure AI-generated code meets our quality standards
# 4. Verify AI did not plagiarize from incompatible licenses

# === INFERENCE/GENERATION ===

# Allow AI systems to read and analyze this repository
ai-inference: allow

# Allow AI code completion tools (GitHub Copilot, Tabnine, etc.)
ai-code-completion: allow

# Allow AI-powered code review and analysis
ai-code-review: allow

# Allow AI to generate documentation from code
ai-documentation: allow
